library(ggplot2)

ggplot(titanic, aes(x = Fare)) +
  geom_density(fill = "lightblue", alpha = 0.5) +
  labs(title = "Density Plot of Fare", x = "Fare", y = "Density")

#The density plot shows the distribution of passenger fares. Most of the fares are between 0 and 50.
  • Histogram
ggplot(titanic, aes(x = Fare)) +
  geom_histogram(binwidth = 10, fill = "steelblue", color = "black") +
  labs(title = "Histogram of Fare", x = "Fare", y = "Frequency")

#The histogram shows that a large number of passengers paid fares under 50, with far fewer paying higher amounts.
